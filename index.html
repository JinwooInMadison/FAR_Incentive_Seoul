function initMap() { // 1. 지도 초기화 map = new naver.maps.Map('map', { center: new naver.maps.LatLng(37.541, 126.986), zoom: 11 }); // 2. 지적편집도 레이어 추가 var cadastralLayer = new naver.maps.CadastralLayer(); cadastralLayer.setMap(map); // 3. 스타일 미리 정의 (데이터가 로드되자마자 입혀짐) map.data.setStyle(function(feature) { return { fillColor: '#ff5722', fillOpacity: 0.35, strokeColor: '#ff5722', strokeWeight: 2, clickable: true }; }); // 4. [교체 핵심] fetch를 이용해 파일을 직접 가져와서 주입 fetch('./Seoul_Planning_68.geojson') .then(function(response) { if (!response.ok) throw new Error('파일 로드 실패: ' + response.status); return response.json(); }) .then(function(json) { map.data.addGeoJson(json); console.log("데이터 주입 성공: 율곡로, 사당 등 포함"); }) .catch(function(error) { console.error("에러 발생:", error); }); // 5. 클릭 이벤트 및 마커 설정 var infowindow = new naver.maps.InfoWindow({ content: '' }); map.data.addListener('click', function(e) { var districtName = e.feature.getProperty('DGM_NM'); var contentString = '
' + '

지구단위계획구역

' + '

이름: ' + districtName + '

' + '
'; infowindow.setContent(contentString); infowindow.open(map, e.coord); }); var marker = new naver.maps.Marker({ position: new naver.maps.LatLng(37.4765, 126.9816), map: map, title: "남현동 602-25" }); }
